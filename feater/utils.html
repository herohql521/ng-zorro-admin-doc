    # 数值转人民币

    /**
 * 转化成RMB元字符串
 * @param number
 * @param digits 当数字类型时，允许指定小数点后数字的个数，默认2位小数
 */
import{yuan} from '@utils'
 money = yuan(200,2)
// ¥200.00

    <div [innerHTML]="money"></div>
<!-- ¥ 200.00 -->

    # 全角转半角

    /**
 *  @param {string}
 */
import{ToCDB} from '@utils'
ToCDB('ｑｕａｎｊｉａｏ')
//quanjiao

    # 格式化金额

    import{moneyFormat} from '@utils'
/**
 * @param num  金额
 * @param [format]  需保留的小数，不设置时原样输出
 */
moneyFormat(3000,2)   //  "3,000.00"
